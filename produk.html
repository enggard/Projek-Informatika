# @title print("Pilih operasi:") print("1. Penjumlahan") print("2. Pengurangan") print("3. Perkalian") print("4. Pembagian") #meminta input dari pengguna pilihan=input("masukkan pilihan 1/2/3/4= ") if pilihan in ("1", "2", "3", "4" ): bil1=float(input("Masukkan bilangan pertama= ")) bil2=float(input("Masukkan bilangan kedua= ")) if pilihan == "1": tambah=bil1+bil2 print("Hasil Penjumlahan=", tambah) elif pilihan =="2": kurang=bil1-bil2 print("Hasil Pengurangan=",kurang) elif pilihan =="3": kali=bil1*bil2 print("Hasil Perkalian=",kali) elif pilihan=="4": bagi=bil1/bil2 print("Hasil Pembagian=",bagi) else: print("tidak valid")
